# Executable MR Evidence Gap Closure Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Close the three highest-risk evidence gaps: one runnable node-permutation asset path, two nontrivial MR cards, and rubric decisions that visibly accept, reject, defer, or mark OOD-stress candidates.

**Architecture:** Use TDD around small JSON assets and Python standard-library runners. The runnable path is an asset-level execution path over a deterministic fixture, not a real MeshGraphNets SUT run, because the current environment exposes no `METBENCH_MGN_DATA_ROOT`, `METBENCH_MGN_REPO`, or `METBENCH_MGN_CHECKPOINT`. SUT execution must remain `blocked` or `no-run` until those artifacts exist.

**Tech Stack:** Python standard library, JSON MR cards, JSON ledgers, Markdown evidence notes, existing `unittest` validation flow.

---

## Evidence Boundaries

- Do not claim MeshGraphNets checkpoint inference unless a worker actually runs a real model from a recorded dataset/checkpoint.
- The node-permutation runner may claim only: transformation code, inverse mapping, metric computation, and verdict ledger serialization work on a deterministic fixture.
- Mirror-y and divergence/conservation cards may claim only design-time rubric decisions unless transformation code and SUT execution exist.
- Rubric decisions must distinguish `retained-design-time`, `retained-ood-stress`, `deferred`, `downgraded-to-diagnostic`, and `rejected-design-time`.
- Every new conclusion must point to a local file, JSON field, command output, or explicit blocked marker.

Create `tools/run_mr_asset.py`:

```python
from __future__ import annotations

import json
from math import sqrt
from pathlib import Path
from typing import Any


def load_json(path: Path) -> dict[str, Any]:
    return json.loads(path.read_text(encoding="utf-8"))


def apply_permutation(values: list[Any], permutation: list[int]) -> list[Any]:
    return [values[index] for index in permutation]


def inverse_permute(values: list[Any], inverse_permutation: list[int]) -> list[Any]:
    return [values[index] for index in inverse_permutation]


def relative_l2(left: list[list[float]], right: list[list[float]]) -> float:
    numerator = 0.0
    denominator = 0.0
    for left_row, right_row in zip(left, right, strict=True):
        for left_value, right_value in zip(left_row, right_row, strict=True):
            numerator += (left_value - right_value) ** 2
            denominator += right_value ** 2
    if denominator == 0.0:
        return 0.0 if numerator == 0.0 else float("inf")
    return sqrt(numerator) / sqrt(denominator)


def run_node_permutation_fixture(root: Path) -> dict[str, Any]:
    root = Path(root)
    fixture_path = root / "research_assets" / "fixtures" / "node_permutation_case.json"
    card_path = root / "research_assets" / "mr_cards" / "node_permutation_equivariance.json"
    output_path = root / "research_assets" / "runs" / "node_permutation_fixture_verdict.json"

    fixture = load_json(fixture_path)
    card = load_json(card_path)

    restored_follow_up = inverse_permute(
        fixture["y_follow_up_permuted"], fixture["inverse_permutation"]
    )
    value = relative_l2(restored_follow_up, fixture["y_source"])
    threshold = float(card["tolerance"]["threshold"])
    verdict = "pass" if value <= threshold else "fail"

    entry = {
        "run_id": "fixture-node-permutation-run-001",
        "mr_id": fixture["mr_id"],
        "source_case_id": fixture["case_id"],
        "follow_up_case_id": fixture["case_id"] + "-permuted",
        "execution_profile": "fixture-transform-metric-only",
        "sut_execution": "not-run",
        "verdict": verdict,
        "metric": {
            "name": card["metric"]["name"],
            "value": value
        },
        "tolerance": card["tolerance"],
        "evidence_artifact": str(fixture_path.relative_to(root))
    }

    ledger = {
        "ledger_id": "node-permutation-fixture-verdict-ledger",
        "evidence_level": "fixture-transform-metric-only",
        "schema_version": "0.2.0",
        "claim_limitations": "This ledger records a fixture-level asset execution path. It does not report MeshGraphNets inference, SUT reliability, violation rates, or empirical pass/fail rates.",
        "entries": [entry]
    }
    output_path.parent.mkdir(parents=True, exist_ok=True)
    output_path.write_text(json.dumps(ledger, indent=2, ensure_ascii=False) + "\n", encoding="utf-8")
    return entry


if __name__ == "__main__":
    run_node_permutation_fixture(Path(__file__).resolve().parents[1])
```

```python
ALLOWED_CANDIDATE_DECISIONS = {
    "retained-design-time",
    "retained-ood-stress",
    "deferred",
    "downgraded-to-diagnostic",
    "rejected-design-time",
}


def validate_rubric_decision_coverage(root: Path) -> list[dict[str, str]]:
    path = root / "research_assets" / "ledgers" / "candidate_ledger.json"
    data, errors = read_json(path, "rubric_decision_coverage")
    if errors:
        return errors
    if not isinstance(data, dict):
        return [error("rubric_decision_coverage", path, "top-level value must be object")]
    entries = data.get("entries")
    if not isinstance(entries, list):
        return [error("rubric_decision_coverage", path, "entries must be list")]
    decisions = {entry.get("decision") for entry in entries if isinstance(entry, dict)}
    required = {
        "retained-design-time",
        "retained-ood-stress",
        "deferred",
        "rejected-design-time",
    }
    missing = sorted(required - decisions)
    return [
        error("rubric_decision_coverage", path, f"missing decision class: {decision}")
        for decision in missing
    ]
```

Then call it from `validate_repo()`.

## 3. 公共基础设施 `mt/mr_common.py`

所有 MR 脚本依赖的统一封装：

```python
# mt/mr_common.py
"""蜕变测试公共工具：模型加载、推理封装、变换算子库。"""
import os, sys
sys.path.insert(0, os.path.join(os.path.dirname(__file__), '..'))

import copy
import numpy as np
import torch
import torch_geometric.transforms as T
from torch_geometric.data import Data

from dataset import FpcDataset
from model.simulator import Simulator
from utils.utils import NodeType

DEVICE = 'cuda:0' if torch.cuda.is_available() else 'cpu'
TRANSFORMER = T.Compose([T.FaceToEdge(), T.Cartesian(norm=False), T.Distance(norm=False)])


def load_model(ckpt='checkpoints/best_model.pth'):
    """加载训练好的 Simulator 并设为 eval 模式。"""
    sim = Simulator(message_passing_num=15, node_input_size=11,
                    edge_input_size=3, device=DEVICE).to(DEVICE)
    state = torch.load(ckpt, weights_only=False, map_location=DEVICE)
    sim.load_state_dict(state['model_state_dict'])
    sim.eval()
    return sim


def load_dataset(split='test', data_root='data'):
    return FpcDataset(data_root=data_root, split=split)


@torch.no_grad()
def infer_one_step(model, graph):
    """单步推理：返回预测的下一时刻速度 [N, 2]。
    注意 transformer 会修改 graph，所以必须深拷贝。"""
    g = copy.deepcopy(graph)
    g = TRANSFORMER(g)
    g = g.to(DEVICE)
    return model(g, velocity_sequence_noise=None).detach().cpu()


@torch.no_grad()
def rollout(model, graph0, n_steps, mask_fixed_velocity=True):
    """从 graph0 出发自回归推理 n_steps 步。
    返回 velocities: tensor [n_steps+1, N, 2]，含初始帧。"""
    g = copy.deepcopy(graph0).to(DEVICE)
    node_type = g.x[:, 0]
    fixed_mask = ~torch.logical_or(node_type == NodeType.NORMAL,
                                    node_type == NodeType.OUTFLOW)
    init_v = g.x[:, 1:3].clone()
    vs = [init_v.cpu()]
    for _ in range(n_steps):
        g_t = TRANSFORMER(copy.deepcopy(g))
        g_t = g_t.to(DEVICE)
        pred_v = model(g_t, velocity_sequence_noise=None)
        if mask_fixed_velocity:
            pred_v[fixed_mask] = init_v[fixed_mask]  # 入口/壁面保持原值
        g.x[:, 1:3] = pred_v
        vs.append(pred_v.detach().cpu())
    return torch.stack(vs, dim=0)


# ===================== 变换算子库 =====================

def mirror_y(graph):
    """关于 y=0 轴的镜像变换：y → -y，v_y → -v_y。"""
    g = copy.deepcopy(graph)
    g.pos = g.pos.clone()
    g.pos[:, 1] = -g.pos[:, 1]
    g.x = g.x.clone()
    g.x[:, 2] = -g.x[:, 2]      # v_y 分量取反
    g.y = g.y.clone()
    g.y[:, 1] = -g.y[:, 1]
    return g


def scale_geometry_and_velocity(graph, k):
    """同步缩放几何与速度：pos→k*pos, v→k*v。
    在 Re=ρUL/ν 不变前提下，模型内部的相对结构应保持。"""
    g = copy.deepcopy(graph)
    g.pos = g.pos.clone() * k
    g.x = g.x.clone()
    g.x[:, 1:3] = g.x[:, 1:3] * k
    g.y = g.y.clone() * k
    return g


def add_velocity_noise(graph, std=1e-4, seed=0):
    """仅对 NORMAL 节点的速度加高斯噪声。"""
    g = copy.deepcopy(graph)
    torch.manual_seed(seed)
    noise = torch.randn_like(g.x[:, 1:3]) * std
    node_type = g.x[:, 0]
    mask = node_type != NodeType.NORMAL
    noise[mask] = 0
    g.x = g.x.clone()
    g.x[:, 1:3] = g.x[:, 1:3] + noise
    return g
```

## 4. MR 集合（5 类）

每个 MR 给出：物理含义 → 形式化定义 → 输入变换 → 输出约束 → 容差 → 可运行代码模板。

---

### 4.1 MR1 基础恒等

**物理含义**：模型确定性检验——同一输入两次推理应产生完全相同的输出。

**形式化定义**：

$$ \forall x \in \mathcal{X},\quad f(x) = f(x) $$

**输入变换**：恒等变换（同一图两次喂入）。

**输出约束**：$\| f(x_1) - f(x_2) \|_\infty = 0$（机器精度级）。

**容差**：`atol = 0`（必须严格相等，因模型为 eval 模式 + 无 Dropout）。

**优先级**：P1（违反代表存在未控随机源，影响所有后续 MR）。

**代码模板** `mt/mr1_identity.py`：

```python
"""MR1 基础恒等蜕变关系：同输入两次推理结果完全一致。"""
import torch
from mr_common import load_model, load_dataset, infer_one_step

def run(n_samples=20):
    model = load_model()
    ds = load_dataset(split='test')
    fails = []
    for i in range(n_samples):
        g = ds[i * 100]    # 跨轨迹采样
        y1 = infer_one_step(model, g)
        y2 = infer_one_step(model, g)
        diff = (y1 - y2).abs().max().item()
        ok = (diff == 0.0)
        print(f"[MR1] sample {i}: max|y1-y2|={diff:.3e}  {'PASS' if ok else 'FAIL'}")
        if not ok:
            fails.append((i, diff))
    pass_rate = 1.0 - len(fails) / n_samples
    print(f"\nMR1 通过率: {pass_rate*100:.2f}%  (要求 P1 ≥ 95%)")
    return pass_rate, fails

if __name__ == '__main__':
    run()
```

### 4.2 MR2 输入扰动稳定性

**物理含义**：鲁棒性检验——速度场加入微小扰动后，输出变化应被控制在可接受范围内。

**形式化定义**：

$$ \|x' - x\|_2 \leq \delta \;\Rightarrow\; \|f(x') - f(x)\|_2 \leq L \cdot \delta $$

其中 $L$ 为局部 Lipschitz 常数（经验上对训练良好的网络 $L \approx 1{\sim}10$）。

**输入变换**：对 NORMAL 节点速度添加 $\mathcal{N}(0, \sigma^2)$ 高斯噪声，$\sigma = 10^{-4}$。

**输出约束**：

$$ \frac{\|f(x') - f(x)\|_2}{\|f(x)\|_2} \leq \varepsilon_{rel} $$

**容差**：$\varepsilon_{rel} = 5\%$（在小扰动下输出相对变化不超过 5%）。

**优先级**：P2（影响泛化能力）。

**代码模板** `mt/mr2_perturbation.py`：

```python
"""MR2 输入扰动稳定性：小噪声扰动后输出相对变化有界。"""
import torch
from mr_common import load_model, load_dataset, infer_one_step, add_velocity_noise

def run(n_samples=20, std=1e-4, eps_rel=0.05):
    model = load_model()
    ds = load_dataset(split='test')
    fails = []
    for i in range(n_samples):
        g = ds[i * 100]
        y0 = infer_one_step(model, g)
        g_pert = add_velocity_noise(g, std=std, seed=i)
        y1 = infer_one_step(model, g_pert)
        rel = (y1 - y0).norm() / (y0.norm() + 1e-12)
        ok = (rel.item() <= eps_rel)
        print(f"[MR2] sample {i}: rel_err={rel.item():.3e}  {'PASS' if ok else 'FAIL'}")
        if not ok:
            fails.append((i, rel.item()))
    pass_rate = 1.0 - len(fails) / n_samples
    print(f"\nMR2 通过率: {pass_rate*100:.2f}%  (要求 P2 ≥ 90%)")
    return pass_rate, fails

if __name__ == '__main__':
    run()
```

### 4.3 MR3 物理约束嵌入（不可压缩连续性）

**物理含义**：物理一致性检验——不可压缩流体应满足 $\nabla \cdot \mathbf{u} = 0$（质量守恒）。

**形式化定义**（弱形式，三角单元上）：对每个三角形 $K$ 用格林定理：

$$ \int_K \nabla \cdot \mathbf{u} \, dA = \oint_{\partial K} \mathbf{u} \cdot \mathbf{n} \, ds $$

离散化后对每个三角形求边界通量之和。

**输入变换**：无变换，直接对预测速度场作守恒检验。

**输出约束**：单元平均散度的相对范数

$$ \frac{\sum_K |\nabla \cdot \mathbf{u}_K|}{\sum_K \|\mathbf{u}_K\| + \epsilon} \leq \varepsilon_{div} $$

**容差**：$\varepsilon_{div} = 5\%$（数据驱动模型为统计近似，不强求严格守恒）。

**优先级**：P0（违反代表物理谬误）。

**代码模板** `mt/mr3_conservation.py`：

```python
"""MR3 物理约束嵌入（不可压缩连续性）：∇·u ≈ 0。"""
import torch
import numpy as np
from mr_common import load_model, load_dataset, infer_one_step

def cell_divergence(pos, cells, velocity):
    """对每个三角单元用格林定理计算 ∫∇·u dA = ∮u·n ds。
    pos: [N,2], cells: [M,3] (numpy int), velocity: [N,2]
    返回 div [M], cell_area [M]."""
    div, area = [], []
    for tri in cells:
        p = pos[tri]          # [3,2]
        v = velocity[tri]     # [3,2]
        flux = 0.0
        for i in range(3):
            p0, p1 = p[i], p[(i+1) % 3]
            edge = p1 - p0
            # 外法向（顺时针为外）：(dy, -dx)；面积有向量校正
            n = np.array([edge[1], -edge[0]])
            vmid = 0.5 * (v[i] + v[(i+1) % 3])
            flux += float(np.dot(vmid, n))
        # 三角面积（正值）
        a = 0.5 * abs((p[1,0]-p[0,0])*(p[2,1]-p[0,1]) -
                      (p[2,0]-p[0,0])*(p[1,1]-p[0,1]))
        div.append(flux)
        area.append(a)
    return np.array(div), np.array(area)

def run(n_samples=20, eps_div=0.05):
    model = load_model()
    ds = load_dataset(split='test')
    fails = []
    for i in range(n_samples):
        g = ds[i * 100]
        y = infer_one_step(model, g).numpy()      # 预测速度 [N,2]
        pos = g.pos.numpy()
        cells = g.face.numpy().T                  # [M,3]
        div, area = cell_divergence(pos, cells, y)
        vmag = np.linalg.norm(y, axis=1)
        denom = vmag.mean() * area.sum() + 1e-12
        rel_div = np.sum(np.abs(div)) / denom
        ok = rel_div <= eps_div
        print(f"[MR3] sample {i}: rel_div={rel_div:.3e}  {'PASS' if ok else 'FAIL'}")
        if not ok:
            fails.append((i, rel_div))
    pass_rate = 1.0 - len(fails) / n_samples
    print(f"\nMR3 通过率: {pass_rate*100:.2f}%  (要求 P0 ≥ 99.9%)")
    return pass_rate, fails

if __name__ == '__main__':
    run()
```

### 4.4 MR4 对称性保持

**物理含义**：圆柱关于 $y=0$ 轴几何对称、入流均匀，则瞬时统计平均流场应满足 $u(x,-y) = u(x,y),\; v(x,-y) = -v(x,y)$。

> 注意：圆柱绕流的瞬时场会出现卡门涡街交替脱落，**单帧**不对称。本 MR 测试两种形态：
> - **形式 A（弱）**：对单帧做镜像变换后输入模型，输出应等于模型对原帧输出的镜像变换。这是"模型对几何对称性的等变性"，与流场是否对称无关。
> - **形式 B（强）**：对长时序场取时间平均后，时均场应满足上述对称性。

**形式化定义**（形式 A）：设对称变换算子 $T_{in}$ 为 $y\to -y, v_y\to -v_y$，$T_{out}$ 同形式作用于速度场，则

$$ f(T_{in}(x)) = T_{out}(f(x)) $$

**输出约束**：$\| f(T_{in}(x)) - T_{out}(f(x)) \|_2 / \|f(x)\|_2 \leq \varepsilon_{sym}$

**容差**：$\varepsilon_{sym} = 5\%$（形式 A，数据驱动统计近似）。

**优先级**：P2。

**代码模板** `mt/mr4_symmetry.py`：

```python
"""MR4 对称性保持（形式 A：模型对镜像的等变性）。"""
import torch
from mr_common import load_model, load_dataset, infer_one_step, mirror_y

def run(n_samples=20, eps_sym=0.05):
    model = load_model()
    ds = load_dataset(split='test')
    fails = []
    for i in range(n_samples):
        g = ds[i * 100]
        y_orig = infer_one_step(model, g)              # f(x)
        y_orig_mirrored = y_orig.clone()
        y_orig_mirrored[:, 1] = -y_orig_mirrored[:, 1]  # T_out(f(x))

        g_mirror = mirror_y(g)
        y_mirror = infer_one_step(model, g_mirror)     # f(T_in(x))

        # 镜像后节点顺序未变，可以直接逐元素比较
        diff = (y_mirror - y_orig_mirrored).norm()
        rel = diff / (y_orig.norm() + 1e-12)
        ok = rel.item() <= eps_sym
        print(f"[MR4] sample {i}: rel_sym_err={rel.item():.3e}  {'PASS' if ok else 'FAIL'}")
        if not ok:
            fails.append((i, rel.item()))
    pass_rate = 1.0 - len(fails) / n_samples
    print(f"\nMR4 通过率: {pass_rate*100:.2f}%  (要求 P2 ≥ 90%)")
    return pass_rate, fails

if __name__ == '__main__':
    run()
```

### 4.5 MR5 尺度不变性（基础形式）

**物理含义**：在 Re 保持不变的相似缩放下，速度场应按相同比例缩放。

**形式化定义**：设缩放因子 $k > 0$，$T_{in}^k: (\mathbf{x}, \mathbf{u}) \to (k\mathbf{x}, k\mathbf{u})$，$T_{out}^k(\mathbf{u}) = k\mathbf{u}$，则

$$ f(T_{in}^k(x)) = T_{out}^k(f(x)) $$

**输入变换**：`pos → k*pos, velocity → k*velocity`（同步缩放）。

**输出约束**：$\| f(T_{in}^k(x)) - k f(x) \|_2 / (k \|f(x)\|_2) \leq \varepsilon_{scale}$

**容差**：$\varepsilon_{scale} = 5\%$。

**优先级**：P2。

**代码模板** `mt/mr5_scale_invariance.py`：

```python
"""MR5 尺度不变性（基础形式）：同步缩放几何与速度，输出应按相同比例缩放。"""
import torch
from mr_common import load_model, load_dataset, infer_one_step, scale_geometry_and_velocity

def run(n_samples=20, ks=(0.5, 2.0), eps_scale=0.05):
    model = load_model()
    ds = load_dataset(split='test')
    fails = []
    total = 0
    for i in range(n_samples):
        g = ds[i * 100]
        y0 = infer_one_step(model, g)
        for k in ks:
            total += 1
            g_k = scale_geometry_and_velocity(g, k)
            yk = infer_one_step(model, g_k)
            expected = k * y0
            rel = (yk - expected).norm() / (expected.norm() + 1e-12)
            ok = rel.item() <= eps_scale
            print(f"[MR5] sample {i} k={k}: rel_err={rel.item():.3e}  {'PASS' if ok else 'FAIL'}")
            if not ok:
                fails.append((i, k, rel.item()))
    pass_rate = 1.0 - len(fails) / total
    print(f"\nMR5 通过率: {pass_rate*100:.2f}%  (要求 P2 ≥ 90%)")
    return pass_rate, fails

if __name__ == '__main__':
    run()
```

## 5. Re-St 不变性 MR【重点】

文档 §5.4.2 选定的目标 MR。验证模型在不同雷诺数缩放下涡脱落主频是否仍满足 Strouhal 数标度律。

### 5.1 物理背景

- **斯特劳哈尔数**：$St = f_s L / U$，其中 $f_s$ 为涡脱落主频。
- **雷诺数**：$Re = \rho U L / \mu$。
- **实验标度律**：在亚临界区（$300 < Re < 2\times10^5$）有 $St \approx 0.198(1 - 19.7/Re)$。
- **Re-St 不变性 MR**：若同步缩放 $(L, U)$ 并保持 Re 不变（要求介质粘度同步调整），则缩放前后模型预测的 $St$ 应在阈值内一致。

### 5.2 形式化定义

输入变换：$x' = (kL, kU, \rho, k\mu)$，使 $Re' = Re$。
输出约束：

$$ |St_{model}(x') - St_{model}(x)| \leq 0.002 $$

（绝对偏差阈值，来源于文档表 15）。

### 5.3 实施流程

1. 选取测试集中一条完整轨迹（600 步），自回归 rollout 得到预测速度场。
2. 在尾迹区固定坐标（例如圆柱后 $x \approx 3D$, $y \approx 0$）选取若干**探针点**。
3. 提取探针点的 $v_y$ 时间序列（涡脱落对横向速度脉动最敏感）。
4. 对时间序列做 FFT，取主峰频率 $f_s$。
5. 由 $St = f_s L / U$ 计算 St（$L$、$U$ 取自原始问题尺度）。
6. 对同一轨迹做几何+速度同步缩放（$k=0.5, 2.0$），重复 1–5。
7. 比较 $St$ 是否在 0.002 阈值内一致；与实验标度律 $St \approx 0.198$ 对比。

### 5.4 代码模板 `mt/mr5_re_st.py`

```python
"""MR5 重点：Re-St 不变性。
对同一初始帧分别在 k=1,0.5,2.0 三种缩放下做 rollout，提取尾迹探针 v_y 时间序列，
做 FFT，比较 Strouhal 数偏差是否 ≤ 0.002。"""
import os
import numpy as np
import torch
import matplotlib.pyplot as plt
from scipy.signal import find_peaks

from mr_common import (load_model, load_dataset, rollout,
                       scale_geometry_and_velocity)

# ---------- 配置 ----------
N_STEPS       = 400        # rollout 步数
DT_SECONDS    = 0.01       # cylinder_flow meta.json 中 dt
CYLINDER_D    = 0.0508     # 圆柱直径 L (m), 数据集默认
U_REF         = 1.0        # 参考来流速度 (m/s), 取入口稳态值
ST_TOLERANCE  = 0.002
PROBE_OFFSETS = [(3.0, 0.0), (4.0, 0.3), (5.0, -0.3)]  # (x/D, y/D)
# -------------------------

def pick_probe_indices(pos_np, cylinder_center, D, k=1.0):
    """在缩放后的坐标系中找出最接近探针位置的节点索引。"""
    indices = []
    for ox, oy in PROBE_OFFSETS:
        target = cylinder_center + np.array([ox * D * k, oy * D * k])
        d2 = np.sum((pos_np - target) ** 2, axis=1)
        indices.append(int(np.argmin(d2)))
    return indices

def estimate_strouhal(vy_series, dt, L, U):
    """对 v_y 时序做 FFT，返回主峰频率对应的 St 与主峰能量占比。"""
    sig = vy_series - vy_series.mean()
    n = len(sig)
    win = np.hanning(n)
    spec = np.abs(np.fft.rfft(sig * win))
    freqs = np.fft.rfftfreq(n, d=dt)
    # 排除 0 频
    idx = np.argmax(spec[1:]) + 1
    f_s = freqs[idx]
    main_ratio = (spec[idx] ** 2) / (spec[1:] ** 2).sum()
    St = f_s * L / U
    return St, main_ratio, freqs, spec

def estimate_cylinder_center(pos_np, node_type_np):
    """从 OBSTACLE 节点估计圆柱中心。"""
    from utils.utils import NodeType
    mask = node_type_np.flatten() == NodeType.OBSTACLE
    if mask.sum() == 0:
        return np.array([0.5, 0.2])  # 经验默认
    return pos_np[mask].mean(axis=0)

def run_one(model, graph, k, results_dir):
    g_k = scale_geometry_and_velocity(graph, k) if k != 1.0 else graph
    velocities = rollout(model, g_k, N_STEPS).numpy()    # [T+1, N, 2]
    pos = g_k.pos.numpy()
    node_type = g_k.x.numpy()[:, 0:1]
    center = estimate_cylinder_center(pos, node_type)
    probes = pick_probe_indices(pos, center, CYLINDER_D, k=k)

    sts, ratios = [], []
    for p_idx in probes:
        vy = velocities[:, p_idx, 1]
        # 缩放后的物理量：dt 不变，L=k*D, U=k*U_REF
        St, main_ratio, freqs, spec = estimate_strouhal(
            vy, DT_SECONDS, CYLINDER_D * k, U_REF * k
        )
        sts.append(St); ratios.append(main_ratio)
        # 频谱图存档
        fig, ax = plt.subplots()
        ax.loglog(freqs[1:], spec[1:])
        ax.axvline(St * (U_REF * k) / (CYLINDER_D * k), color='r', ls='--',
                   label=f'St={St:.3f}')
        ax.set_xlabel('f (Hz)'); ax.set_ylabel('|V_y(f)|'); ax.legend()
        fig.savefig(os.path.join(results_dir, f'k{k}_probe{p_idx}_fft.png'))
        plt.close(fig)
    return np.mean(sts), np.mean(ratios), sts

def run(rollout_index=0, ks=(1.0, 0.5, 2.0)):
    os.makedirs('mt/results', exist_ok=True)
    model = load_model()
    ds = load_dataset(split='test')
    # 取该轨迹的第一帧
    n_per_tra = ds.num_sampes_per_tra
    g0 = ds[rollout_index * n_per_tra]
    St_base = None
    rows = []
    for k in ks:
        St_mean, main_ratio, sts = run_one(model, g0, k, 'mt/results')
        if k == 1.0:
            St_base = St_mean
        deviation = None if St_base is None else abs(St_mean - St_base)
        passed = (k == 1.0) or (deviation <= ST_TOLERANCE)
        rows.append({
            'k': k, 'St_mean': St_mean, 'main_peak_ratio': main_ratio,
            '|St-St_base|': deviation, 'pass': passed,
            'sts_per_probe': sts,
        })
        print(f"[Re-St] k={k}: St={St_mean:.4f}  "
              f"main_peak_ratio={main_ratio:.2%}  "
              f"|ΔSt|={deviation}  {'PASS' if passed else 'FAIL'}")
    # 写报告
    with open('mt/results/re_st_report.md', 'w') as f:
        f.write('| k | St_mean | main_peak_ratio | |ΔSt| | pass |\n')
        f.write('|---|---|---|---|---|\n')
        for r in rows:
            f.write(f"| {r['k']} | {r['St_mean']:.4f} | "
                    f"{r['main_peak_ratio']:.2%} | {r['|St-St_base|']} | "
                    f"{r['pass']} |\n")
    return rows

if __name__ == '__main__':
    run()
```

## 8. 常见问题与排查

| 现象 | 可能原因 | 排查 |
|---|---|---|
| `FaceToEdge` 报 `Expected face with shape [3,M] or [4,M]` | `cells` 维度不对 | 打印 `g.face.shape`，确认是 `[3, M]` |
| 自有数据加载维度对不上 | 数据非三角网格 / 命名不符合 `{split}.npz` 约定 | 参考 §2.2 字段要求，先在原版 cylinder_flow 跑通 |
| MR1 不通过 | 模型存在未控随机源（如 dropout 未关、batch norm 训练模式） | 确认 `model.eval()`；检查 `Simulator.training==False` |
| MR3 散度过大 | 模型未学到不可压缩约束 / 探针单元在边界 | 排除 OBSTACLE/INFLOW 邻接单元；增大训练数据 |
| Re-St 主峰能量占比 < 85% | 探针位置不在尾迹区 / rollout 步数不足 | 增大 `N_STEPS`，调整 `PROBE_OFFSETS` |
| 缩放后 St 偏离过大 | 模型对训练 Re 范围外推失败 | 与文档表 15 的低 Re 失效机制比对 |

## 9. 致学生的几点提醒

1. **先跑通基线再做蜕变测试**。如果 `rollout.py` 生成的 GIF 与 ground truth 视觉差异巨大，则任何 MR 失效都无法判断是模型本身的物理缺陷还是基线未收敛。
2. **不要轻易断言"代码与数据不匹配"**。`FaceToEdge` 同时支持 2D 三角和 3D 四面体，cylinder_flow 数据集本身就是 2D 三角网格，仓库 README 也明确说支持自有数据。出现"维度对不上"先打印 shape 排查，再下结论。
3. **MR 的容差不是越严越好**。数据驱动模型本质是统计近似，文档 §3.3.3 明确说明"预测近似满足守恒律"，故 P0 类容差也应留出工程余量。
4. **失败本身就是有价值的实验结果**。文档表 15 的 4 个衍生用例有 2 个未通过（衍生 2 和衍生 4），并通过物理归因揭示了模型对低 Re 的局限。学生应以同样态度记录失败案例。
